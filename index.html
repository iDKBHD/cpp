Project 1: Swap Numbers

using namespace std;

	int a,b,temp;

	cin >> b;

Project 2: Calculating Salary of an employee given his basic pay. Calculate salary of the employee. Let HRA be "10%" of basic pay and TA be "5%" of basic pay. Let employees pay professional tax as "2%" of total salary. Calculate salary payable after deductions.

#include <iostream>
using namespace std;

	float basic_pay, HRA, TA, total_salary, professional_tax, net_salary;

	cout << "Enter the basic pay: ";
	cin >> basic_pay;

	HRA = 0.10*basic_pay;
	TA = 0.05*basic_pay;

	total_salary = basic_pay + HRA + TA;

	professional_tax = 0.02*total_salary;

	net_salary = total_salary - professional_tax;

	cout << "Basic Pay: " << basic_pay << endl;
	cout << "HRA (10% of Basic Pay): " << HRA << endl;
	cout << "TA (5% of Basic Pay): " << TA << endl;
	cout << "Total Salary before deductions: " << total_salary << endl;
	cout << "Professional Tax (2% of Total Salary): " << professional_tax << endl;
	cout << "net Payable Salary: " << net_salary << endl;

	return 0;

}

Project 4: Armstrong Number

#include <iostream>
using namespace std;

int main() {
    int num, originalNum, remainder, cube, result=0;

    cout << "Enter a Three Digit Number: ";
    cin >> num;

    originalNum = num;

    while (originalNum != 0) {
        remainder = originalNum % 10;
        cube = remainder * remainder * remainder;
        result = result + cube;
        originalNum /= 10;
    }

    if (result == num) {
        cout << num << " is an Armstrong Number." << endl;
    }

    else {
        cout << num << " is not an Armstrong Number" << endl;
    }

    result 0;
}

Project 6: Palindrome

#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int main() {
    string str, reversedStr;

    cout << "Enter a String: ";
    cin >> str;

    reversedStr = str;

    reverse (reversedStr.begin(), reversedStr.end());

    if (str == reversedStr) {
        cout << str << " is a Palindrome." << endl;
    }

    else {
        cout << str << " is not a Palindrome." << endl;
    }

    return 0;
}

Project 7: Vowels and Consonants

#include <iostream>
#include <string>

using namespace std;

int main() {
    string str;
    int vowels = 0, consonants = 0;

    cout << "Enter a String: ";
    getline(cin, str);

    for (int i = 0; i < str.length(); i++) {
        char ch = tolower(str[i]);
        if ((ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')) {
            vowels++;
        } else if ((ch >= 'a' && ch <= 'z')) {
            consonants++;
        }
    }

    cout << "No. of Vowels: " << vowels << endl;
    cout << "No. of Consonants: " << consonants << endl;

    return 0;
}

Project 8: Factorial (with a function)

#include <iostream>
using namespace std;

int factorial (int n) {
    if (n <= 0) {
        return -1;
    }

    int fact = 1;
    
    for (int i=1; i<=n; i++) {
        fact *= i;
    }

    return fact;
}

int main () {
    int num;

    cout << "Enter no. of which you want Factorial of: ";
    cin >> num;

    int result = factorial(num);

    if (result != -1) {
        cout << "The Factorial is: " << result << endl;
    }
    else {
        cout << "Factorial is not defined: " << endl;
    }

    return 0;
}

Project 9: Prime number or not

#include <iostream>
using namespace std;

bool prime(int n) {
    if (n<0) {
        return false;
    }

    for (int i=2; i <= n/2; i++) {
        if (n % 1 == 0) {
            return false;
        }
    }

    return true;
}

int main() {
    int num;

    cout << "Enter a no.: " << endl;
    cin >> num;

    if (prime(num)) {
        cout << "Number is Prime" << endl;
    }

    else {
        cout << "Number is not Prime" << endl;
    }

    return 0;
}
